  170.   function check_signature(&$request, $consumer, $token, $signature) {
  171.     $decoded_sig = base64_decode($signature);
  172.  
  173.     $base_string = $request->get_signature_base_string();
  174.  
  175.     // Fetch the public key cert based on the request
  176.     $cert = $this->fetch_public_cert($request);

  179.     $publickeyid = openssl_get_publickey($cert);
  180.  
  181.     // Check the computed signature against the one passed in the query

  184.     // Release the key resource

  212.    * attempt to build up a request from what was passed to the server
  213.    */
  214.   static function from_request($http_method=NULL, $http_url=NULL, $parameters=NULL) {
  215.     $scheme = (!isset($_SERVER['HTTPS']) || $_SERVER['HTTPS'] != "on") ? 'http' : 'https';
  216.     @$http_url or $http_url = $scheme . '://' . $_SERVER['HTTP_HOST'] . ':' . $_SERVER['SERVER_PORT'] . $_SERVER['REQUEST_URI'];
  217.     @$http_method or $http_method = $_SERVER['REQUEST_METHOD'];
  218.  
  219.     $request_headers = OAuthRequest::get_headers();
  220.  
  221.     // let the library user override things however they'd like, if they know
  222.     // which parameters to use then go for it, for example XMLRPC might want to
  223.     // do this
  224.     if ($parameters) {
  225.       $req = new OAuthRequest($http_method, $http_url, $parameters);
  226.     } else {
  227.       // collect request parameters from query string (GET) and post-data (POST) if appropriate (note: POST vars have priority)
  228.       // NOTE: $_GET and $_POST will strip duplicate query parameters
  229.       $req_parameters = $_GET;
  230.       if ($http_method == "POST" && @strstr($request_headers["Content-Type"], "application/x-www-form-urlencoded") ) {
  231.         $req_parameters = array_merge($req_parameters, $_POST);
  232.       }
  233.  
  234.       // next check for the auth header, we need to do some extra stuff
  235.       // if that is the case, namely suck in the parameters from GET or POST
  236.       // so that we can include them in the signature
  237.       if (@substr($request_headers['Authorization'], 0, 6) == "OAuth ") {
  238.         $header_parameters = OAuthRequest::split_header($request_headers['Authorization']);
  239.         $parameters = array_merge($req_parameters, $header_parameters);
  240.         $req = new OAuthRequest($http_method, $http_url, $parameters);
  241.       } else $req = new OAuthRequest($http_method, $http_url, $req_parameters);
  242.     }
  243.  
  244.     return $req;
  245.   }

  293.       parse_str($query_str, $php_parsed_query);
  294.  
  295.       if(OAuthUtil::oauth_http_build_query($parsed_query) != OAuthUtil::oauth_http_build_query($php_parsed_query)) {

  432.   /**

  434.    * parameters, has to do some unescaping
  435.    */
  436.   static function split_header($header) {
  437.     $pattern = '/(([-_a-z]*)=("([^"]*)"|([^,]*)),?)/';
  438.     $offset = 0;
  439.     $params = array();
  440.     while (preg_match($pattern, $header, $matches, PREG_OFFSET_CAPTURE, $offset) > 0) {
  441.       $match = $matches[0];
  442.       $header_name = $matches[2][0];
  443.       $header_content = (isset($matches[5])) ? $matches[5][0] : $matches[4][0];
  444.       $params[$header_name] = OAuthUtil::urldecode_rfc3986( $header_content );
  445.       $offset = $match[1] + strlen($match[0]);
  446.     }
  447.  
  448.     if (isset($params['realm'])) {
  449.        unset($params['realm']);
  450.     }
  451.  
  452.     return $params;
  453.   }

  773. class OAuthUtil {
  774.  
  775.   // This function takes a input like a=b&a=c&d=e and returns the parsed
  776.   // parameters like this
  777.   // array('a' => array('b','c'), 'd' => 'e')
  778.   static function parse_parameters( $input ) {
  779.     if (!isset($input) || !$input) return array();
  780.  
  781.     $pairs = explode('&', $input);
  782.  
  783.     $parsed_parameters = array();
  784.     foreach ($pairs as $pair) {
  785.       $split = explode('=', $pair, 2);
  786.       $parameter = OAuthUtil::urldecode_rfc3986($split[0]);
  787.       $value = isset($split[1]) ? OAuthUtil::urldecode_rfc3986($split[1]) : '';
  788.  
  789.       if (isset($parsed_parameters[$parameter])) {
  790.         // We have already recieved parameter(s) with this name, so add to the list
  791.         // of parameters with this name
  792.  
  793.         if (is_scalar($parsed_parameters[$parameter])) {
  794.           // This is the first duplicate, so transform scalar (string) into an array
  795.           // so we can add the duplicates
  796.           $parsed_parameters[$parameter] = array($parsed_parameters[$parameter]);
  797.         }
  798.  
  799.         $parsed_parameters[$parameter][] = $value;
  800.       } else {
  801.         $parsed_parameters[$parameter] = $value;
  802.       }
  803.     }
  804.     return $parsed_parameters;
  805.   }
  806.  
  807.   // Utility function for turning the Authorization: header into
  808.   // parameters, has to do some unescaping
  809.   // Can filter out any non-oauth parameters if needed (default behaviour)
  810.   static function split_header($header, $only_allow_oauth_parameters = true) {
  811.     $pattern = '/(([-_a-z]*)=("([^"]*)"|([^,]*)),?)/';
  812.     $offset = 0;
  813.     $params = array();
  814.     while (preg_match($pattern, $header, $matches, PREG_OFFSET_CAPTURE, $offset) > 0) {
  815.       $match = $matches[0];
  816.       $header_name = $matches[2][0];
  817.       $header_content = (isset($matches[5])) ? $matches[5][0] : $matches[4][0];
  818.       if (preg_match('/^oauth_/', $header_name) || !$only_allow_oauth_parameters) {
  819.         $params[$header_name] = OAuthUtil::urldecode_rfc3986($header_content);
  820.       }
  821.       $offset = $match[1] + strlen($match[0]);
  822.     }
  823.  
  824.     if (isset($params['realm'])) {
  825.        unset($params['realm']);
  826.     }
  827.  
  828.     return $params;
  829.   }

  858.   static function build_http_query($params) {
  859.     if (!$params) return '';
  860.  
  861.     // Urlencode both keys and values
  862.     $keys = OAuthUtil::urlencode_rfc3986(array_keys($params));
  863.     $values = OAuthUtil::urlencode_rfc3986(array_values($params));
  864.     $params = array_combine($keys, $values);
  865.  
  866.     // Parameters are sorted by name, using lexicographical byte value ordering.
  867.     // Ref: Spec: 9.1.1 (1)
  868.     uksort($params, 'strcmp');
  869.  
  870.     $pairs = array();
  871.     foreach ($params as $parameter => $value) {
  872.       if (is_array($value)) {
  873.         // If two or more parameters share the same name, they are sorted by their value
  874.         // Ref: Spec: 9.1.1 (1)
  875.         natsort($value);
  876.         foreach ($value as $duplicate_value) {
  877.           $pairs[] = $parameter . '=' . $duplicate_value;
  878.         }
  879.       } else {
  880.         $pairs[] = $parameter . '=' . $value;
  881.       }
  882.     }
  883.     // For each parameter, the name is separated from the corresponding value by an '=' character (ASCII code 61)
  884.     // Each name-value pair is separated by an '&' character (ASCII code 38)
  885.     return implode('&', $pairs);
  886.   }

  888.   static function oauth_parse_string($query_string) {
  889.     if(!isset($query_string)) {
  890.       return array();
  891.     }
  892.     $pairs = explode('&', $query_string);
  893.     $query_arr = array();
  894.     foreach($pairs as $pair) {
  895.       list($k, $v) = explode('=', $pair, 2);
  896.  
  897.       // Handle duplicate query params
  898.       if(isset($query_arr[$k])) {
  899.  
  900.         // Transform scalar to array
  901.         if(is_scalar($query_arr[$k])) {
  902.           $query_arr[$k] = array($query_arr[$k]);
  903.         }
  904.         $query_arr[$k][] = $v;
  905.       } else {
  906.         $query_arr[$k] = $v;
  907.       }
  908.     }
  909.     return $query_arr;
  910.   }

  912.   static function oauth_http_build_query($params) {
  913.  
  914.     $out = false;
  915.     if(!empty($params))
  916.     {
  917.       // Urlencode both keys and values
  918.       $keys = OAuthUtil::urlencode_rfc3986(array_keys($params));
  919.       $values = OAuthUtil::urlencode_rfc3986(array_values($params));
  920.       $params = array_combine($keys, $values);
  921.  
  922.       // Sort by keys (natsort)
  923.       uksort($params, 'strcmp');
  924.  
  925.       $pairs = array();
  926.       foreach ($params as $k => $v) {
  927.         if (is_array($v)) {
  928.           natsort($v);
  929.           foreach ($v as $duplicate_value) {
  930.             $pairs[] = $k . '=' . $duplicate_value;
  931.           }
